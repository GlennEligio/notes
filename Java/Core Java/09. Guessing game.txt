
        Random random = new Random();
        Scanner scanner = new Scanner(System.in);
        int guessNumber = random.nextInt(30);
        int tries = 4;
        int tryAnswer;
        String name;
        boolean correct = false;

        System.out.println("Welcome to the 'Guess the Number' game");
        System.out.println("Now, tell us your name, player.");
        name = scanner.next();
        System.out.println("Now give your first us your first guess, " + name);
        tryAnswer = scanner.nextInt();


        for (int i = 0; i < tries; i++){
            if (tryAnswer > guessNumber){
                System.out.println("Wrong, your guess is higher than the answer.");
                tryAnswer = scanner.nextInt();
            }else if (tryAnswer < guessNumber){
                System.out.println("Wrong, your guess is lower than the answer");
                tryAnswer = scanner.nextInt();
            }else if (tryAnswer == guessNumber){
                correct = true;
                break;
            }
        }

        if (correct){
            System.out.println("Congratulations");
        }else{
            System.out.println("Game Over");
        }
