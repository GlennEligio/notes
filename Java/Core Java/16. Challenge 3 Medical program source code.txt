package com.glenneligio.medicalprogram;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {

        int option;
        Eye leftEye = new Eye ("Left eye", "Short sighted", "Blue");
        Eye rightEye = new Eye ("Right eye", "Short sighted", "Blue");
        Heart heart = new Heart ("Heart", "Normal", 65);
        Stomach stomach = new Stomach ("Stomach", "PUD", false);
        Skin skin = new Skin ("Skin", "Burned");
        Person patient = new Person(null, 0, leftEye, rightEye, heart, stomach, skin);

        Scanner scanner = new Scanner(System.in);
        Prompt prompt;

        //introduction prompt
        System.out.println("Enter your name: ");
        patient.setName(scanner.next());
        System.out.println("Enter your age: ");
        patient.setAge(scanner.nextInt());
        prompt = new Prompt (patient.getName(), patient.getAge());

        //main menu prompt
        do{
            prompt.mainPrompt(patient.getName(), patient.getAge());
            option = prompt.promptOption();

            switch (option) {
                case 1 -> leftEye.promptOrgan();
                case 2 -> rightEye.promptOrgan();
                case 3 -> heart.promptOrgan();
                case 4 -> stomach.promptOrgan();
                case 5 -> skin.promptOrgan();
            }
        }while(option < 6);

        if (option >= 6){
            System.out.println("Program closed.");
        }



    }
}

Eye class (child class of Organ)

package com.glenneligio.medicalprogram;

import java.util.Scanner;

public class Eye extends Organ{
    private String color;
    private boolean isClose = true;

    public Eye(String name, String condition, String color) {
        super(name, condition);
        this.color = color;
        this.isClose = isClose;
    }

    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public boolean isClose() {
        return isClose;
    }

    public void setClose(boolean close) {
        isClose = close;
    }

    public void isClosed(){
        if (isClose()) System.out.println(getName() + " is close.");
        else System.out.println(getName() + " is open.");
    }

    @Override
    public void promptOrgan(){
        System.out.println("Name: " + getName());
        System.out.println("Medical condition: " + getCondition());
        System.out.println("Color: " + getColor());

        if (!isClose()) {
            System.out.println("\t1. Close the eye");
        }else{
            System.out.println("\t1. Open the eye");
        }

        Scanner scanner = new Scanner(System.in);
        int eyeOption = scanner.nextInt();

        if (eyeOption == 1){
            if(isClose()){
                setClose(false);
                System.out.println(getName() + " is now opened");
            }else if(!isClose()){
                setClose(true);
                System.out.println(getName() + " is now closed");
            }
        }
    }
}
